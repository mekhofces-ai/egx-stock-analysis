PS C:\Users\omar.mokhtar\Documents\New project> function Get-DotEnvValue {
    param([string]$Key)

    if (!(Test-Path $EnvPath)) {
        return $null
    }

    $line = Get-Content $EnvPath | Where-Object {
        $_ -match "^\s*$([regex]::Escape($Key))\s*="
    } | Select-Object -Last 1

    if (!$line) {
        return $null
    }

    return ($line -replace "^\s*$([regex]::Escape($Key))\s*=\s*", "").Trim().Trim('"').Trim("'")
}

    param(
        [string]$Key,
        [string]$Value
    )

    $escapedKey = [regex]::Escape($Key)
    $newLine = "$Key=$Value"

    $content = @()
    if (Test-Path $EnvPath) {
        $content = Get-Content $EnvPath -ErrorAction SilentlyContinue
    }

    $found = $false

    $newContent = $content | ForEach-Object {
        if ($_ -match "^\s*$escapedKey\s*=") {
            $found = $true
            $newLine
        } else {
            $_
        }
    }

    if (!$found) {
        $newContent += $newLine
    }

    Set-Content -Path $EnvPath -Value $newContent -Encoding UTF8
}
PS C:\Users\omar.mokhtar\Documents\New project> function Find-TelegramBotTokenInProject {
    $excludePattern = "\\(\.git|\.venv|venv|env|__pycache__|node_modules|\.streamlit)\\"

    $files = Get-ChildItem -Path $ProjectPath -Recurse -File -ErrorAction SilentlyContinue |
        Where-Object {
            $_.FullName -notmatch $excludePattern -and
            $_.Extension -in ".env", ".py", ".txt", ".json", ".toml", ".yaml", ".yml", ".ini", ".cfg"
        }

    foreach ($file in $files) {
        try {
            $matches = Select-String -Path $file.FullName -Pattern "([0-9]{6,}:[A-Za-z0-9_\-]{20,})" -ErrorAction SilentlyContinue
            foreach ($m in $matches) {
                if ($m.Matches.Count -gt 0) {
                    return $m.Matches[0].Groups[1].Value
                }
            }
        } catch {}
    }

    return $null
}
